### Content Generation Automation

**AI-Powered Caption Generator:**
```python
import anthropic # or google.generativeai
import json
import random

def generate_social_captions(song_title, artist_name, stream_count, platform):
    """
    Generate 50 caption variations using Claude/Gemini
    """
    client = anthropic.Anthropic(api_key="YOUR_KEY")
    
    prompt = f"""Generate 10 engaging social media captions for {platform} 
    promoting the song "{song_title}" by {artist_name}. 
    Current stream count: {stream_count}
    
    Requirements:
    - Mix of emotional, vulnerable, hype, and storytelling angles
    - Include relevant hashtags (max 10 for IG, 5 for TikTok)
    - Add call-to-action to stream on Spotify
    - Vary tone: some casual, some professional
    - Make it authentic and engaging
    
    Return as JSON array of caption objects."""
    
    message = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=2000,
        messages=[{"role": "user", "content": prompt}]
    )
    
    return json.loads(message.content[0].text)
```
